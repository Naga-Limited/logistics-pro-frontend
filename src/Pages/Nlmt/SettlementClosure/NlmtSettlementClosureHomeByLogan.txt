                              (row) => String(row.Advance_Given) === String(selectedCheckPers)
                            )

                      setRowData(filteredRows)
                      setSearchFilterData(filteredSearch)
                    }}
                  >
                    Filter
                  </CButton>
                  <CButton
                    size="lg-sm"
                    color="warning"
                    className="m-3 px-3 text-white"
                    onClick={(e) => {
                      exportToCSV()
                    }}
                  >
                    Export
                  </CButton>
                </CCol>
              </CRow>

              {/* Payment Submission Section */}
              <CRow className="mt-3">
                <CCol xs={12} md={6}>
                  <CFormLabel htmlFor="tripPaymentSubmission">
                    Select Tripsheets for Payment Submission
                  </CFormLabel>
                  <TripPaymentSubmissionComponent
                    size="sm"
                    name="tripPaymentSubmission"
                    id="tripPaymentSubmission"
                    selectedValue={values.tripPaymentSubmission}
                    onFocus={onFocus}
                    onBlur={onBlur}
                    onChange={handleMultipleChange}
                    isMultiple={true}
                    label="Select Tripsheets"
                    noOptionsMessage="No Tripsheets Available"
                    search_type="payment_submission"
                    // search_data={searchFilterData}
                    search_data={searchFilterData}
                  />
                </CCol>

                {values.tripPaymentSubmission && values.tripPaymentSubmission.length > 0 && (
                  <CCol className="mt-4" xs={12} sm={9} md={3}>
                    <CButton
                      color="success"
                      className="m-2 text-white"
                      onClick={() => {
                        checkModalDisplay()
                      }}
                    >
                      Confirm Payment Submission
                    </CButton>

                    <CModal
                      alignment="center"
                      size="xl"
                      visible={assignPSModal}
                      onClose={() => setAssignPSModal(false)}
                    >
                      <CModalHeader>
                        <CModalTitle>{'NLMT Tripsheets : Payment Submission'}</CModalTitle>
                      </CModalHeader>

                      <CModalBody>
                        {tripPaymentsArray.length > 0 ? (
                          <CTable>
                            {singleVendorArray.length > 0 && singleVendorArray[0].vendor_info && (
                              <CRow className="mb-3">
                                <CCol md={3}>
                                  <CFormLabel htmlFor="vname">Vendor Name</CFormLabel>
                                  <CFormInput
                                    style={{ fontWeight: 'bolder' }}
                                    name="vname"
                                    size="sm"
                                    id="vname"
                                    value={singleVendorArray[0].vendor_info.owner_name || ''}
                                    readOnly
                                  />
                                </CCol>

                                <CCol md={3}>
                                  <CFormLabel htmlFor="vcode">Vendor Code</CFormLabel>
                                  <CFormInput
                                    style={{ fontWeight: 'bolder' }}
                                    name="vcode"
                                    size="sm"
                                    id="vcode"
                                    value={singleVendorArray[0].vendor_info.vendor_code || ''}
                                    readOnly
                                  />
                                </CCol>

                                <CCol md={3}>
                                  <CFormLabel htmlFor="shedname">Shed Name</CFormLabel>
                                  <CFormInput
                                    style={{ fontWeight: 'bolder' }}
                                    name="shedname"
                                    size="sm"
                                    id="shedname"
                                    value={
                                      singleVendorArray[0].vendor_info.nlmt_shed_info?.shed_name ||
                                      ''
                                    }
                                    readOnly
                                  />
                                </CCol>

                                <CCol md={3}>
                                  <CFormLabel htmlFor="remarks">Remarks</CFormLabel>
                                  <CFormInput
                                    size="sm"
                                    name="remarks"
                                    id="remarks"
                                    value={message}
                                    onChange={handleChangeRemarks}
                                  />
                                </CCol>
                              </CRow>
                            )}

                            <CTableHead className="mt-3" style={{ background: 'pink' }}>
                              <CTableRow>
                                <CTableHeaderCell scope="col" style={{ width: '5%' }}>
                                  S.No
                                </CTableHeaderCell>
                                <CTableHeaderCell scope="col" style={{ width: '15%' }}>
                                  Tripsheet No.
                                </CTableHeaderCell>
                                <CTableHeaderCell scope="col" style={{ width: '15%' }}>
                                  Vehicle No.
                                </CTableHeaderCell>
                                <CTableHeaderCell scope="col" style={{ width: '15%' }}>
                                  Driver Name
                                </CTableHeaderCell>
                                <CTableHeaderCell scope="col" style={{ width: '15%' }}>
                                  Route
                                </CTableHeaderCell>
                                <CTableHeaderCell scope="col" style={{ width: '15%' }}>
                                  Actual Freight{' '}
                                </CTableHeaderCell>
                                <CTableHeaderCell scope="col" style={{ width: '15%' }}>
                                  Advance Amount{' '}
                                </CTableHeaderCell>
                                <CTableHeaderCell scope="col" style={{ width: '15%' }}>
                                  Deduction Amount{' '}
                                </CTableHeaderCell>
                                <CTableHeaderCell scope="col" style={{ width: '15%' }}>
                                  Balance Freight
                                </CTableHeaderCell>
                              </CTableRow>
                            </CTableHead>
                            <CTableBody>
                              {tripPaymentsArray.map((val, ind) => {
                                // console.log(val, 'tripPaymentsArraytripPaymentsArray')
                                const freightRate =
                                  val.tripsheet_info?.trip_vehicle_route?.freight_rate &&
                                  val.vehicle_assignment?.[0]?.billed_qty
                                    ? val.tripsheet_info.trip_vehicle_route.freight_rate *
                                      Number(val.vehicle_assignment[0].billed_qty)
                                    : 0
                                const deductionAmt =
                                  val.trip_settlement_info?.diversion_return_charges

                                const settlementExpenses =
                                  Number(val.trip_settlement_info?.expense || 0) -
                                  Number(val.advance_payment_info?.advance_payment || 0)
                                const actualFreightRate =
                                  val.advance_payment_info?.actual_freight || '0'
                                const advanceAmt = val.advance_payment_info?.advance_payment || '0'
                                return (
                                  <CTableRow key={ind}>
                                    <CTableDataCell scope="col" style={{ width: '5%' }}>
                                      {ind + 1}
                                    </CTableDataCell>
                                    <CTableDataCell scope="col" style={{ width: '15%' }}>
                                      {val.tripsheet_info?.nlmt_tripsheet_no || '-'}
                                    </CTableDataCell>
                                    <CTableDataCell scope="col" style={{ width: '15%' }}>
                                      {val.vehicle_info?.vehicle_number || '-'}
                                    </CTableDataCell>
                                    <CTableDataCell scope="col" style={{ width: '15%' }}>
                                      {val.driver_name || '-'}
                                    </CTableDataCell>
                                    <CTableDataCell scope="col" style={{ width: '15%' }}>
                                      {val.tripsheet_info?.trip_vehicle_route?.route_name || '-'}
                                    </CTableDataCell>
                                    <CTableDataCell scope="col" style={{ width: '15%' }}>
                                      {freightRate}
                                    </CTableDataCell>
                                    <CTableDataCell scope="col" style={{ width: '15%' }}>
                                      {advanceAmt}
                                    </CTableDataCell>
                                    <CTableDataCell scope="col" style={{ width: '15%' }}>
                                      {deductionAmt}
                                    </CTableDataCell>
                                    <CTableDataCell scope="col" style={{ width: '15%' }}>
                                      {settlementExpenses}
                                    </CTableDataCell>
                                  </CTableRow>
                                )
                              })}
                              {/* Summary Footer */}
                              <CTableRow
                                style={{
                                  backgroundColor: '#f8f9fa',
                                  borderTop: '2px solid #dee2e6',
                                }}
                              >
                                <CTableDataCell colSpan={8} style={{ padding: '12px 15px' }}>
                                  <div
                                    style={{
                                      display: 'flex',
                                      justifyContent: 'space-between',
                                      alignItems: 'center',
                                      width: '100%',
                                      fontSize: '15px',
                                    }}
                                  >
                                    {/* Left Side → TS Count */}
                                    <div style={{ fontWeight: '600' }}>
                                      TS Count :
                                      <span
                                        style={{
                                          color: '#0d6efd',
                                          fontWeight: 'bold',
                                          marginLeft: '5px',
                                        }}
                                      >
                                        {tripPaymentsArray.length}
                                      </span>
                                    </div>

                                    {/* Right Side → Totals */}
                                    <div
                                      style={{
                                        display: 'flex',
                                        gap: '25px',
                                        fontWeight: '600',
                                      }}
                                    >
                                      <div>
                                        Total Freight :
                                        <span
                                          style={{
                                            color: '#0d6efd',
                                            fontWeight: 'bold',
                                            marginLeft: '5px',
                                          }}
                                        >
                                          {tripPaymentsArray.reduce(
                                            (sum, val) =>
                                              sum +
                                              Number(val.advance_payment_info?.actual_freight || 0),
                                            0
                                          )}
                                        </span>
                                      </div>

                                      <div>
                                        Total Advance :
                                        <span
                                          style={{
                                            color: '#0d6efd',
                                            fontWeight: 'bold',
                                            marginLeft: '5px',
                                          }}
                                        >
                                          {tripPaymentsArray.reduce(
                                            (sum, val) =>
                                              sum +
                                              Number(
                                                val.advance_payment_info?.advance_payment || 0
                                              ),
                                            0
                                          )}
                                        </span>
                                      </div>

                                      <div>
                                        Total Balance :
                                        <span
                                          style={{
                                            color: 'green',
                                            fontWeight: 'bold',
                                            marginLeft: '5px',
                                          }}
                                        >
                                          {tripPaymentsArray.reduce(
                                            (sum, val) =>
                                              sum +
                                              (Number(val.trip_settlement_info?.expense || 0) -
                                                Number(
                                                  val.advance_payment_info?.advance_payment || 0
                                                )),
                                            0
                                          )}
                                        </span>
                                      </div>
                                    </div>
                                  </div>
                                </CTableDataCell>
                              </CTableRow>
                            </CTableBody>
                          </CTable>
                        ) : (
                          <>Tripsheets Not Found / Invalid</>
                        )}
                      </CModalBody>

                      <CModalFooter>
                        <CButton
                          color="primary"
                          style={{ marginRight: '2%' }}
                          onClick={() => {
                            setAssignPSModal(false)
                            setFetch(false)
                            submitPayment()
                          }}
                        >
                          {'Submit'}
                        </CButton>
                        <CButton
                          color="primary"
                          onClick={() => {
                            setAssignPSModal(false)
                          }}
                        >
                          {'Cancel'}
                        </CButton>
                      </CModalFooter>
                    </CModal>
                  </CCol>
                )}
              </CRow>

              <CustomTable
                columns={columns}
                pagination={false}
                data={rowData}
                fieldName={'Driver_Name'}
                showSearchFilter={true}
              />
            </CContainer>
          </CCard>
        </>
      )}
    </>
  )
}

export default NlmtSettlementClosureHome
